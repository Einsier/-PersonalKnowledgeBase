    class Solution {
    private int[][] nears = {  // æ¯ä¸ªä¸‹æ ‡å¯¹åº”çš„é‚»å±…ä¸‹æ ‡
        {1, 3},
        {0, 2, 4},
        {1, 5},
        {0, 4},
        {1, 3, 5},
        {2, 4}
    };
    private Set<String> inq = new HashSet<>();  // è®°å½•è¯¥çŠ¶æ€æ˜¯å¦å…¥é˜Ÿ
    public int slidingPuzzle(int[][] board) {
        StringBuilder start = new StringBuilder();  // äºŒç»´æ•°ç»„åŒ–ä¸ºå­—ç¬¦ä¸²å½¢å¼
        for(int i = 0; i < 6; ++i)  start.append(board[i/3][i%3]);
        return bfs(start.toString(), "123450"); 
    }
    // å•å‘BFS
    int bfs(String start, String target) {
        int res = 0;
        Queue<String> q = new LinkedList<>();
        q.offer(start);
        inq.add(start);
        while(!q.isEmpty()) {
            for(int i = 0, len = q.size(); i < len; ++i) {
                if(q.peek().equals(target)) return res;
                List<String> nexts = getNexts(q.poll());
                for(String next : nexts) {
                    q.offer(next);
                    inq.add(next);
                }
            }
            ++res;
        }
        return -1;
    }
    // è·å–å½“å‰boardç§»åŠ¨ä¸€æ­¥èƒ½å½¢æˆçš„boardsï¼ˆä¸”æœªæ›¾å…¥é˜Ÿï¼‰
    List<String> getNexts(String board) {
        List<String> res = new ArrayList<>();
        int zero = board.indexOf("0");  // è·å–0çš„ä½ç½®
        for(int near : nears[zero]) {
            String temp = moveZero(board, zero, near);
            if(!inq.contains(temp)) res.add(temp);
        }
        return res;
    }
    // ç§»åŠ¨0çš„ä½ç½®
    String moveZero(String board, int zero, int near) {
        StringBuilder sb = new StringBuilder(board);
        sb.setCharAt(zero, board.charAt(near));
        sb.setCharAt(near, '0');
        return sb.toString();
    }
}
