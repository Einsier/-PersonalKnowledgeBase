class Solution {

    bool canJump(vector<int>& nums) {
        if(nums.size() == 1) return true;
        int maxReach = nums[0];
        for(int i = 1; i < nums.size(); ++i) {
            if(i > maxReach) return false;  // å½“å‰ä½ç½®ä¸å¯è¾¾
