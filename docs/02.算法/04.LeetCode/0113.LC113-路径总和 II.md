class Solution {
public:
    vector<vector<int>> res;
    vector<vector<int>> pathSum(TreeNode* root, int targetSum) {
        dfs(root, targetSum, {});
        return res;
    }
    void dfs(TreeNode* root, int targetSum, vector<int> path) {
        if(root) {
            path.push_back(root->val);  // åœ¨è·¯å¾„ä¸­æ·»åŠ èŠ‚ç‚¹

        }
    }
};
