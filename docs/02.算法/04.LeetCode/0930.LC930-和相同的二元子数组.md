class Solution {
public:
    int numSubarraysWithSum(vector<int>& nums, int goal) {
        int res = 0, curSum = 0;
        unordered_map<int, int> map;
        map[0] = 1;
        for(auto num : nums) {
            curSum += num;
            int target = curSum - goal;
            if(map.find(target) != map.end()) res += map[target];
            map[curSum]++;
        }
        return res;
    }
};
