class NumArray {
public:
    vector<int> preSum;
    NumArray(vector<int>& nums) {
        preSum = vector<int>(nums.size() + 1, 0);
        for(int i = 0; i < nums.size(); ++i) {
            preSum[i+1] = preSum[i] + nums[i];
        }
    }
    
    int sumRange(int i, int j) {
        return preSum[j+1] - preSum[i];
    }
};
