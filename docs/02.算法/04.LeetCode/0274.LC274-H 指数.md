class Solution {
public:
    int hIndex(vector<int>& citations) {
        int left = 0, right = citations.size(), mid;
        while(left < right) {
            mid = (left + right + 1) / 2;
            int count = 0;  // ç»Ÿè®¡å¤§äºç­‰äºhçš„å…ƒç´ ä¸ªæ•°

                if(c >= mid) ++count;

            if(mid <= count) {
                left = mid;

            }
        }

    }
};
