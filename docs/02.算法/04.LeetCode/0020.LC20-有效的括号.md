class Solution {
public:
    bool isValid(string s) {
        stack<char> left;
        for(char c : s) {
            if(c == '(' || c == '{' || c == '[') {
                left.push(c);
            }else if(c == ')') {
                if(!left.empty() && left.top() == '(') {
                    left.pop();
                }else return false;
            }else if(c == '}') {
                if(!left.empty() && left.top() == '{') {
                    left.pop();
                }else return false;
            }else {
                if(!left.empty() && left.top() == '[') {
                    left.pop();
                }else return false;
            }
        }
        return left.empty();  // å·¦æ‹¬å·å¤šä½™ä¹Ÿæ— æ•ˆ
    }
};
